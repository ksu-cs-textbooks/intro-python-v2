Loops - Keys

dict_3 = {
    "cat": "mammal",
    "lizard": "reptile",
    "goldfish": "fish",
    "chickadee": "bird"
}
for key in dict_3:
    print(f"{key}: {dict_3[key]}")

Loops - Keys & Values

dict_3 = {
    "cat": "mammal",
    "lizard": "reptile",
    "goldfish": "fish",
    "chickadee": "bird"
}
for key, value in dict_3.items():
    print(f"{key}: {value}")
